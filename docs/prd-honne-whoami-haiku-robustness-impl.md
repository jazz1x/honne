### B.2 `honne_py/axis.py` 신규 시그니처

```python
# scripts/honne_py/axis.py
from pathlib import Path
from typing import Literal
import json, sys
from . import extract, io as honne_io

from typing import Optional  # Python 3.9 호환 — PEP 604 (`str | None`) 금지

AXES = ("lexicon", "reaction", "workflow", "obsession", "ritual", "antipattern")
LOCALES = ("ko", "en", "jp")

# axis → extract 함수 매핑 (B.1에서 발췌)
_EXTRACTORS = {
    "lexicon":     extract.extract_lexicon,
    "reaction":    extract.extract_reaction,
    "workflow":    extract.extract_workflow,
    "obsession":   extract.extract_obsession,
    "ritual":      extract.extract_ritual,
    "antipattern": extract.extract_antipattern,
}

def _top_k(items: list[dict], k: int = 3) -> list[dict]:
    """결정론 정렬:
       1) 빈도(frequency or support_count) desc
       2) ts asc
       3) session_id asc (최종 tiebreak)
       중복 text는 첫 등장만 유지(dedup by normalized text).
       k개 미만이면 있는 만큼 반환(0~2개 가능).
    """
    seen = set(); dedup = []
    for it in items:
        t = (it.get("text") or it.get("quote") or "").strip()
        if not t or t in seen: continue
        seen.add(t); dedup.append(it)
    dedup.sort(key=lambda x: (
        -(x.get("frequency") or x.get("support_count") or 0),
        x.get("ts") or "",
        x.get("session_id") or "",
    ))
    return dedup[:k]

def _load_template(locale: str, axis: str) -> dict:
    """templates/axes.<locale>.md 파서. 파싱 규칙(불변식):
       1) `## <axis_key>` 헤더로 섹션 식별 (정확 매치)
       2) 섹션 내 각 줄 `^- <key>: <value>$` (정규식), value는 rstrip('\\n')
       3) 필수 키 집합: {label, hitl_question, report_header, connective}
       4) 누락 시 KeyError (run()에서 exit 2로 전환)
       5) connective는 따옴표 내 문자열 그대로; 리드/트레일 공백 보존
    """
    import re
    root = Path(__file__).parent.parent.parent / f"skills/whoami/templates/axes.{locale}.md"
    txt = root.read_text(encoding="utf-8")
    m = re.search(rf"(?m)^## {re.escape(axis)}$\n((?:- .+\n?)+)", txt)
    if not m: raise KeyError(f"axis section missing: {axis}")
    kv = {}
    for line in m.group(1).splitlines():
        mm = re.match(r"- ([a-z_]+): (.*)$", line)
        if mm:
            v = mm.group(2)
            # connective 값은 "quoted string" 형태이므로 벗겨낸다
            if v.startswith('"') and v.endswith('"'):
                v = v[1:-1]
            kv[mm.group(1)] = v
    required = {"label", "hitl_question", "report_header", "connective"}
    missing = required - kv.keys()
    if missing: raise KeyError(f"{axis}: missing {missing}")
    return kv

def _render_block(out: dict, tpl: dict) -> str:
    """HITL 텍스트 블록 조립 (B.7 참조)."""
    # 구현: B.7 스펙 따름
    ...

def run(name: str, locale: str, scan_path: Path,
        emit_hitl_block: bool = False) -> int:
    if name not in AXES: return 2
    if locale not in LOCALES:
        sys.stderr.write(f"locale must be one of {LOCALES}\n"); return 2
    if not scan_path.exists(): return 66

    # 1. extract_*로 축별 시그널 추출 (counters, top_examples)
    tmp = scan_path.parent / f".axis_{name}.json"
    _EXTRACTORS[name](input_path=scan_path, out_path=tmp)
    signal = honne_io.load_cache(tmp)  # dict: {axis, counters, top_examples, ...}

    # 2. 시그널을 키로 scan.json 원문 메시지에서 quote 수집 (B.6 계약)
    quotes = collect_quotes(scan_path, name, signal, k=3)

    tpl = _load_template(locale, name)
    if not quotes:
        out = {"axis": name, "quotes": [], "candidate_claim": None,
               "insufficient_evidence": True}
    else:
        candidate = tpl["connective"].join(q["text"] for q in quotes)
        out = {"axis": name, "quotes": quotes, "candidate_claim": candidate,
               "insufficient_evidence": False}

    if emit_hitl_block:
        sys.stdout.write(_render_block(out, tpl))
    else:
        json.dump(out, sys.stdout, ensure_ascii=False)
    return 0

def validate(text: str, locale: str,
             skip_if_overlaps: Optional[str] = None) -> int:
    """exit codes:
         0 통과
         2 forbidden phrase 매치 / 길이 < 8 / locale 불명 / forbidden.json 파싱 실패
         3 skip_if_overlaps 매치 (past rejection 재출현) — Step 3 reframe 필터용
    """
    import json
    if locale not in LOCALES: return 2
    root = Path(__file__).parent.parent.parent / "skills/whoami/templates/forbidden.json"
    try:
        forbidden = json.loads(root.read_text())[locale]
    except Exception:
        return 2
    if len(text.strip()) < 8: return 2
    for phrase in forbidden:
        if phrase in text: return 2
    if skip_if_overlaps and skip_if_overlaps.strip() and skip_if_overlaps.strip() in text:
        return 3
    return 0
```

### B.6 축별 Quote 수집 계약 (`collect_quotes`)

**문제**: `extract_*` 산출은 `{axis, counters, top_examples, ...}` 집계 dict이며 원문 quote·session_id·ts를 포함하지 않음. `lexicon`만 top_examples에 `{word, sessions}` 있고 나머지 5축은 `top_examples=[]`. 따라서 quote 수집은 **axis.py 책임**이며 scan.json 원문을 2차 순회한다.

**축별 키 소스 표** (counter/top_examples → scan.json 매칭):

| axis | key source | scan.json 매칭 규칙 (user-role 메시지만) |
|---|---|---|
| lexicon | `top_examples[].word` (최대 k개) | `word`가 공백/경계 포함 정확히 등장하는 첫 메시지 |
| reaction | `counters` 키 (상위 k개, 값 desc) | 해당 reaction 키 regex (redact된 원문에 대해) 첫 매치 메시지 |
| workflow | `counters` 키 (상위 k개) | 키 문자열을 substring으로 포함하는 첫 메시지 |
| obsession | `top_preambles[].text` (최대 k개) | preamble text가 앞부분에 등장하는 첫 메시지 |
| ritual | `counters` 키 (상위 k개) | 세션당 첫 user 메시지 중 키 매치 |
| antipattern | `counters` 키 (상위 k개) | 키 substring 포함 첫 메시지 |

**시그니처**:
```python
def collect_quotes(scan_path: Path, axis: str, signal: dict, k: int = 3) -> list[dict]:
    """축별 키를 뽑아 scan.json의 user-role 메시지에서 first-match quote로 변환.

    반환 요소 스키마 (불변식):
      {"session_id": str, "text": str, "ts": str, "frequency": int, "key": str}

    결정론 규칙:
      1) 키 후보 정렬 = 빈도 desc, 키 사전순 asc (동점 tiebreak)
      2) session 순회 = scan.json sessions 원본 순서 유지 (입력 순서 = ts asc)
      3) 한 키당 첫 매치 세션의 첫 매치 메시지 1건만 채택
      4) text 길이 > 200자 → 첫 200자 + U+2026 (단일 문자 "…", locale 불문 고정)
      5) 동일 text가 두 키에서 나오면 먼저 처리된 키만 유지(dedup by text)
      6) 결과 길이 ≤ k. 키 부족 시 0~k-1개 반환(insufficient_evidence 판단은 run()이)
      7) 순회 중 예외 → 해당 키 skip, 남은 키로 계속
    """
```

**구현 요점** (Sonnet은 이 요점대로만 코딩, 추가 설계 금지):
- `scan = json.loads(scan_path.read_text())`
- 축별 키 추출은 위 표. obsession은 `signal.get("top_preambles", [])` 접근.
- user-role 필터: `msg["role"] == "user"`.
- regex 사용은 reaction만 — 축별 패턴은 `extract.extract_reaction`의 기존 패턴과 1:1 대응 (extract.py에서 상수로 export 받아 재사용; 신규 regex 작성 금지).
- 빈도는 signal에서 가져온 해당 키의 `counters[key]` 또는 `top_examples[i].sessions` 값.

### B.7 HITL 블록 렌더 스펙 (`_render_block`)

**목표**: locale별 고정 형식의 사용자 표시용 텍스트 블록을 바이트 결정론으로 조립.

**출력 포맷** (UTF-8, LF 종단, trailing newline 1개):
```
{report_header}

{hitl_question}

- [{session_id[:8]}] {ts} — {text}
- [{session_id[:8]}] {ts} — {text}
- [{session_id[:8]}] {ts} — {text}

candidate: {candidate_claim}

[y / n / edit]
```

불변식:
1. `report_header`, `hitl_question`은 템플릿에서 그대로 (의역 금지)
2. quote 리스트는 `collect_quotes` 반환 순서 유지, 없으면 섹션 전체 생략하고 `insufficient_evidence` 문구("[insufficient evidence]")를 그 자리에 한 줄
3. `session_id[:8]`은 고정 슬라이싱, 8자 미만은 그대로
4. 푸터 `[y / n / edit]`는 바이트 고정 (locale 불문)
5. 줄 사이 공백 줄 수 정확히 1
6. `insufficient_evidence=true`면 블록은 `{report_header}\n\n[insufficient evidence]\n`로 단축

골든 픽스처는 이 포맷의 스냅샷.
